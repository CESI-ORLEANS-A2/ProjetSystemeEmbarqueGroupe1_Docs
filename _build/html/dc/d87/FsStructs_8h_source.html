97// Reserved characters for exFAT names and FAT LFN.
98inline bool lfnReservedChar(uint8_t c) {
99 return c < 0X20 || c == '"' || c == '*' || c == '/' || c == ':' || c == '<' ||
100 c == '>' || c == '?' || c == '\\' || c == '|';
101}
102//------------------------------------------------------------------------------
103// Reserved characters for FAT short 8.3 names.
104inline bool sfnReservedChar(uint8_t c) {
105 if (c == '"' || c == '|' || c == '[' || c == '\\' || c == ']') {
106 return true;
107 }
108 // *+,./ or :;<=>?
109 if ((0X2A <= c && c <= 0X2F && c != 0X2D) || (0X3A <= c && c <= 0X3F)) {
110 return true;
111 }
112 // Reserved if not in range (0X20, 0X7F).
113 return !(0X20 < c && c < 0X7F);
114}
