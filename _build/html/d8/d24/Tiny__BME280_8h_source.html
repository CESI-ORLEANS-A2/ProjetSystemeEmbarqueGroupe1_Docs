21/*
22 Tiny_BME280 Modification
23
24 This is a forked and minimized library for the BME280 sensor. It will provide
25 functions to read temperature, humidity and pressure using I2C communication.
26 This libary is based on the more robust Adafruit_BME280_Library but with SPI
27 and other features removed to reduce PROGMEM space. This library is ideal for
28 smaller controllers like the ATtiny85.
29
30 Adafruit_BME280 - https://github.com/adafruit/Adafruit_BME280_Library
31
32*/

288 struct config {
289 // inactive duration (standby time) in normal mode
290 // 000 = 0.5 ms
291 // 001 = 62.5 ms
292 // 010 = 125 ms
293 // 011 = 250 ms
294 // 100 = 500 ms
295 // 101 = 1000 ms
296 // 110 = 10 ms
297 // 111 = 20 ms
298 unsigned int t_sb : 3;
299
300 // filter settings
301 // 000 = filter off
302 // 001 = 2x filter
303 // 010 = 4x filter
304 // 011 = 8x filter
305 // 100 and above = 16x filter
306 unsigned int filter : 3;
307
308 // unused - don't set
309 unsigned int none : 1;
310
312 unsigned int get() { return (t_sb << 5) | (filter << 2) ; }
313 };
314 config _configReg;

321 struct ctrl_meas {
322 // temperature oversampling
323 // 000 = skipped
324 // 001 = x1
325 // 010 = x2
326 // 011 = x4
327 // 100 = x8
328 // 101 and above = x16
329 unsigned int osrs_t : 3;
330
331 // pressure oversampling
332 // 000 = skipped
333 // 001 = x1
334 // 010 = x2
335 // 011 = x4
336 // 100 = x8
337 // 101 and above = x16
338 unsigned int osrs_p : 3;
339
340 // device mode
341 // 00 = sleep
342 // 01 or 10 = forced
343 // 11 = normal
344 unsigned int mode : 2;
345
347 unsigned int get() { return (osrs_t << 5) | (osrs_p << 2) | mode; }
348 };
349 ctrl_meas _measReg;
